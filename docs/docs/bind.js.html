/**
 * @function bind
 * @description Bind a model to elements on the page. When the model is updated, set the elements to the updated values.
 * @param props {object} The model to watch for updates 
 * @param callback {function} A method to call when the model has been updated 
 * @returns Proxy
 */
const bind = (props, callback) => {
    let handler = {
        set: function (target, key, value) {
            target[key] = value;
            if (typeof callback === 'function') {
                callback();
            }

            if (typeof callback === 'undefined') {
                const elements = document.querySelectorAll(`.${key}[data-bind="true"]`);
                for (let element of elements) {
                    if (element instanceof HTMLInputElement || element instanceof HTMLSelectElement || element instanceof HTMLTextAreaElement) {
                        element.value = target[key];
                    } else {
                        element.textContent = target[key];
                    }
                }
            }
            return true;
        },
        get: function (target, key) {
            return target[key];
        }
    }

    return new Proxy(props, handler);
}

export { bind };
